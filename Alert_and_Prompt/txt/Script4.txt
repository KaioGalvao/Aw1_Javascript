        function insertionSort(array) {
            for (let i = 1; i < array.length; i++) {
                let valorAtual = array[i];
                let j;
                for (j = i - 1; j >= 0 && array[j] > valorAtual; j--) {
                    array[j + 1] = array[j];
                }
                array[j + 1] = valorAtual;
            }
            return array;
        }
